Rails API: People Controller

# app/controllers/api/v1/people_controller.rb
class Api::V1::PeopleController < Api::V1::BaseController
  before_action :authenticate_user!
  before_action :set_person, only: [:show, :update, :destroy]

  def index
    @people = current_user.people.includes(:relationships, :media)
    render json: PeopleSerializer.new(@people).serialized_json
  end

  def create
    @person = current_user.people.build(person_params)
    
    if @person.save
      create_relationship if relationship_params.present?
      render json: PersonSerializer.new(@person).serialized_json, status: :created
    else
      render json: { errors: @person.errors.full_messages }, status: :unprocessable_entity
    end
  end

  def update
    if @person.update(person_params)
      render json: PersonSerializer.new(@person).serialized_json
    else
      render json: { errors: @person.errors.full_messages }, status: :unprocessable_entity
    end
  end

  private

  def person_params
    params.require(:person).permit(:first_name, :last_name, :gender, 
                                   :date_of_birth, :date_of_death, :is_deceased)
  end

  def set_person
    @person = current_user.people.find(params[:id])
  end
end

Relationship Calculator: Core Algorithm

// improvedRelationshipCalculator.js - Core relationship logic
export function calculateRelationshipToRoot(person, root, allPeople, relationships) {
  const personId = parseInt(person.id);
  const rootId = parseInt(root.id);
  
  if (personId === rootId) return 'Self';
  
  // Build relationship maps
  const { childToParents, spouseMap } = buildRelationshipMaps(relationships);
  
  // Check direct relationships first
  const directRelation = checkDirectRelationship(personId, rootId, childToParents, spouseMap);
  if (directRelation) return directRelation;
  
  // Check step relationships
  const stepRelation = checkStepRelationship(personId, rootId, childToParents, spouseMap, allPeople);
  if (stepRelation) return stepRelation;
  
  // Check extended family (uncles, cousins, etc.)
  const extendedRelation = checkExtendedFamily(personId, rootId, childToParents, allPeople);
  if (extendedRelation) return extendedRelation;
  
  return 'Unrelated';
}

function checkDirectRelationship(personId, rootId, childToParents, spouseMap) {
  // Parent-child relationships
  const personParents = childToParents.get(personId) || new Set();
  if (personParents.has(rootId)) return 'Parent';
  
  const rootParents = childToParents.get(rootId) || new Set();
  if (rootParents.has(personId)) return 'Child';
  
  // Spouse relationships
  const rootSpouses = spouseMap.get(rootId) || new Set();
  if (rootSpouses.has(personId)) return 'Spouse';
  
  return null;
}

Rails Model: Person with Relationships

# app/models/person.rb
class Person < ApplicationRecord
  belongs_to :user
  has_many :relationships, dependent: :destroy
  has_many :relatives, through: :relationships
  has_many :media, as: :attachable, dependent: :destroy
  has_many :timeline_items, dependent: :destroy
  has_many :facts, dependent: :destroy
  has_one :profile, dependent: :destroy
  
  validates :first_name, presence: true
  validates :last_name, presence: true
  validates :gender, inclusion: { in: %w[Male Female Other] }
  validate :death_date_after_birth_date
  validate :deceased_status_consistency
  
  scope :living, -> { where(is_deceased: false) }
  scope :deceased, -> { where(is_deceased: true) }
  
  def full_name
    "#{first_name} #{last_name}".strip
  end
  
  def age
    return nil unless date_of_birth
    
    end_date = is_deceased && date_of_death ? Date.parse(date_of_death) : Date.current
    birth_date = Date.parse(date_of_birth)
    
    ((end_date - birth_date) / 365.25).floor
  end
  
  def parents
    relationships.where(relationship_type: 'parent').includes(:relative).map(&:relative)
  end
  
  def children
    Person.joins(:relationships)
          .where(relationships: { relative_id: id, relationship_type: 'parent' })
  end
  
  private
  
  def death_date_after_birth_date
    return unless date_of_birth && date_of_death
    
    if Date.parse(date_of_death) <= Date.parse(date_of_birth)
      errors.add(:date_of_death, 'must be after birth date')
    end
  end
  
  def deceased_status_consistency
    if is_deceased && date_of_death.blank?
      errors.add(:date_of_death, 'is required when person is marked as deceased')
    end
  end
end
